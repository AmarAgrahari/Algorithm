create table abc11
(
symbol varchar2(10) primary key,
value1 int
);

insert into abc11 values(?,?)

create or replace procedure insertuser(symbol1 in varchar,value11 in integer)
is
BEGIN
  INSERT INTO abc11 (symbol, value1) VALUES (symbol1,value11);
EXCEPTION
  WHEN DUP_VAL_ON_INDEX THEN
    UPDATE abc11
    SET    value1= value11
    WHERE symbol=symbol1;
END;
/

select * from abc11;
